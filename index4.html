1. Find Largest Number

 #include

}

C++ Programs

 #include

int main()

1. Find Largest Number

 #include

Sequence Programs

1. Print Numbers from 1 to 10

 #include
int main()
{
int i;
for (i = 1; i <= 10; i++)
{
printf("%d ", i);
}
return 0;
}

1. Print Even Numbers from 1 to 20

#include
using namespace std;
int main()
{
int i;
for (i = 2; i <= 20; i += 2)
{
cout << i << " ";
}
return 0;
}

1. Print Fibonacci Series

 #include
using namespace std;
int main()
{
int a = 0, b = 1, c, n = 10;
cout << a << " " << b << " ";
for (int i = 3; i <= n; i++)
{
c = a + b;
cout << c << " ";
a = b;
b = c;
}
return 0;
}
